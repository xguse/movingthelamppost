def make_graph(rounds=1):
    graph = nx.Graph()
    r = 0
    while rounds-r > 0:
        # complete
        graph.add_edge('A%s.1' % (r), 'B%s.1' % (r))
        graph.add_edge('B%s.1' % (r), 'C%s.1' % (r))
        graph.add_edge('C%s.1' % (r), 'A%s.1' % (r))
        #complete
        graph.add_edge('A%s.2' % (r), 'B%s.2' % (r))
        graph.add_edge('B%s.2' % (r), 'C%s.2' % (r))
        graph.add_edge('C%s.2' % (r), 'A%s.2' % (r))
        #complete
        graph.add_edge('A%s.3' % (r), 'B%s.3' % (r))
        graph.add_edge('B%s.3' % (r), 'C%s.3' % (r))
        graph.add_edge('C%s.3' % (r), 'A%s.3' % (r))
        # incomplete
        graph.add_edge('A%s.4' % (r), 'B%s.4' % (r))
        graph.add_edge('B%s.4' % (r), 'C%s.4' % (r))
        r+=1
    return graph

def get_completes_nx(graph):
    subgraphs = nx.connected_component_subgraphs(graph)
    completes = [x for x in subgraphs if len(x.edges()) ==3]
    return completes
